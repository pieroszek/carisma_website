
// disable and enable scroll

function disableScroll() {
    // Get the current scroll position
    var scrollPosition = window.scrollY || window.pageYOffset;
    // Save the current scroll position
    document.body.dataset.scroll = scrollPosition;
    // Set the body to fixed position
    document.body.style.position = 'fixed';
    // Set the body width to 100% to prevent horizontal scrolling
    document.body.style.width = '100%';
    // Set the body top to the negative of the scroll position to keep it in place
    document.body.style.top = '-' + scrollPosition + 'px';
}

function enableScroll() {
    // Get the previous scroll position
    var scrollPosition = document.body.dataset.scroll;
    // Reset the body position to static
    document.body.style.position = 'static';
    // Reset the body width
    document.body.style.width = 'auto';
    // Reset the body top
    document.body.style.top = 'auto';
    // Scroll to the previous position
    window.scrollTo(0, scrollPosition);
}
